        :konverter til Array;
        :tell opp rød-grønn;
        :tell opp borgerlig;
        :tell opp andre;
        if (borg > rg)
          :biggest = borg;

        endif
        :vis biggest;

Testplan

  • Sjekk at en bare kan skrive inn prosenter mellom 0 og 100
    Koden mangler sjekker foreløpig
  • Sjekk at koden summerer riktig for hver blokk
    Manuell sjekk viser at det stemmer
    En test i koden sjekker om tellopp virker PASSED
    En test i koden sjekker om søyler lages riktig PASSED

    Manuell test viser at det er i orden

Test funksjoner i blokker.js

Jeg har lagt inn testene under i blokker.js. De kjøres hver gang skjema lastes og skriver ut testdata i console 

    assert(
      tellOpp(
        [
        { dataset: { id: "a" }, value: 12 },
        { dataset: { id: "b" }, value: 33 },
        { dataset: { id: "c" }, value: 6 }
        ],
        ["a", "b"]
      ) === 45,
      "opptelling blokk 1"
    );
    
    assert(
      tellOpp(
        [
        { dataset: { id: "a" }, value: 12 },
        { dataset: { id: "b" }, value: 33 },
        { dataset: { id: "c" }, value: 6 }
        ],
        ["c"]
      ) === 6,
      "opptelling blokk 2"
    );
    
    assert(
      lageSoyle(3, "blue").style.backgroundColor === "blue",
      "lagSoyle er blå"
    );
    
    assert(
      lageSoyle(3, "blue").style.backgroundColor !== "red",
      "lagSoyle er ikke rød"
    );
    
    assert(
      lageSoyle(33, "blue").style.width === "99px",
